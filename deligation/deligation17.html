import java.applet.Applet;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.JFrame;

class ColorChangeApplet extends Applet implements ActionListener { Color[] colors = {Color.red, Color.green, Color.blue};
int colorIndex = 0;

public void init() {
// Create and add buttons
Button redButton = new Button("Red");
redButton.addActionListener(this);
add(redButton);

Button greenButton = new Button("Green");
greenButton.addActionListener(this);
add(greenButton);

Button blueButton = new Button("Blue");
blueButton.addActionListener(this);
add(blueButton);
}

public void paint(Graphics g) {
g.setColor(colors[colorIndex]);
g.fillRect(0, 0, getWidth(), getHeight());
}

public void actionPerformed(ActionEvent e) {
Button button = (Button) e.getSource();

if (button.getLabel().equals("Red")) {
colorIndex = 0;
} else if (button.getLabel().equals("Green")) {
colorIndex = 1;
} else if (button.getLabel().equals("Blue")) {
colorIndex = 2;
}

repaint();
}

public static void main(String[] args) {
JFrame frame = new JFrame("Color Change Applet");
ColorChangeApplet applet = new ColorChangeApplet();
applet.setSize(400, 300);
frame.add(applet);

frame.pack();
frame.setVisible(true);
frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
}
}
